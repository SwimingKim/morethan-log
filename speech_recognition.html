r = sr.Recognizer()
with sr.Microphone() as source:
    print('listening...')
    audio = r.listen(source, timeout=10, phrase_time_limit=10)
    print("......")

try:
    text_google = r.recognize_google(audio, language='ko', show_all=True)
    text_google = dict(text_google)['alternative'][0]['transcript'] if 'alternative' in dict(text_google).keys() else ""
    text_vosk = r.recognize_vosk(audio, language='ko')
    text_vosk = json.loads(text_vosk)['text']
    text_whisper = r.recognize_whisper(audio, language='ko')
    print("[Google]", text_google)
    print("[Vosk]", text_vosk)
    print("[whisper]", text_whisper)
except sr.UnknownValueError:
    print("Recognizer Failed..")
except sr.RequestError as e:
    print("Request Failed...", e)
